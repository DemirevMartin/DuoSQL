CREATE TABLE _dict (
    name character varying(20),
    dict dictionary
);

CREATE TABLE drives (
    id integer,
    person text,
    color text,
    car text,
    _sentence bdd
);

CREATE TABLE saw (
    id integer,
    witness text,
    color text,
    car text,
    _sentence bdd
);

---------------------

CREATE TABLE person_certain (
    id integer PRIMARY KEY,
    person text
);

CREATE TABLE car_certain (
    id integer PRIMARY KEY,
    color text,
    car text
);

CREATE TABLE drives_certain (
    person_id integer REFERENCES person_certain(id),
    car_id integer REFERENCES car_certain(id)
);

CREATE TABLE saw_certain (
    person_id integer REFERENCES person_certain(id),
    car_id integer REFERENCES car_certain(id)
);
